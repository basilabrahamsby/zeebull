server {
    server_name www.teqmates.com teqmates.com;
    client_max_body_size 50M;

    # 1. Landing Page (Root)
    location / {
        root /var/www/html;
        index index.html;
        try_files $uri $uri/ /index.html;
    }

    # 2. User App (Frontend)
    location /pomma {
        alias /var/www/pomma;
        index index.html;
        try_files $uri $uri/ /pomma/index.html;
    }

    # 3. Dashboard (Admin)
    location /pommaadmin {
        alias /var/www/pommaadmin;
        index index.html;
        try_files $uri $uri/ /pommaadmin/index.html;
    }

    # 4. Backend API
    location /api {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    # Swagger Docs
    location /docs {
        proxy_pass http://127.0.0.1:8000/docs;
    }
    location /openapi.json {
        proxy_pass http://127.0.0.1:8000/openapi.json;
    }

    # Enable Gzip Compression
    gzip on;
    gzip_vary on;
    gzip_min_length 10240;
    gzip_proxied expired no-cache no-store private auth;
    gzip_types text/plain text/css text/xml text/javascript application/x-javascript application/xml;
    gzip_disable "MSIE [1-6]\.";

    # Handle uploads access via /pomma/uploads/ (legacy)
    location ^~ /pomma/uploads/ {
        alias /opt/pomma/ResortApp/uploads/;
        autoindex off;
        expires 30d;
        add_header Cache-Control "public, no-transform";
    }

    # NEW: Orchid Files (Media)
    # Serves content from /var/www/inventory/ResortApp/uploads/
    location /orchidfiles/ {
        alias /var/www/inventory/ResortApp/;
        try_files $uri $uri/ =404;
        add_header Cache-Control "public, max-age=31536000";
        autoindex off;
    }

    location /inventoryfiles/ {
        alias /var/www/inventory/ResortApp/static/;
        try_files $uri $uri/ =404;
        add_header Cache-Control "public, max-age=31536000";
    }

    # Serving Uploaded Files directly via /uploads/ (legacy)
    location /uploads/ {
        alias /opt/pomma/ResortApp/uploads/;
        autoindex off;
        expires 30d;
        add_header Cache-Control "public, no-transform";
    }

    # Fix for frontend expecting /pommaapi path
    location /pommaapi/ {
        rewrite ^/pommaapi/(.*) /$1 break;
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    listen 443 ssl; # managed by Certbot

    location /orchidapi/ {
        proxy_pass http://127.0.0.1:8011/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
        add_header Cache-Control "no-cache, no-store, must-revalidate";
    }

    location /orchid/ {
        alias /var/www/html/orchid/;
        try_files $uri $uri/ /orchid/index.html =404;
        add_header Cache-Control "no-cache, no-store, must-revalidate";
        add_header Pragma "no-cache";
        add_header Expires "0";
    }

    # Redirect /orchid/admin to /orchid/admin/
    location = /orchid/admin {
        return 301 /orchid/admin/;
    }

    location /orchid/admin/ {
        alias /var/www/html/orchidadmin/;
        try_files $uri $uri/ /orchid/admin/index.html;
        add_header Cache-Control "no-cache, must-revalidate";
    }

    ssl_certificate /etc/letsencrypt/live/teqmates.com/fullchain.pem; # managed by Certbot
    ssl_certificate_key /etc/letsencrypt/live/teqmates.com/privkey.pem; # managed by Certbot
    include /etc/letsencrypt/options-ssl-nginx.conf; # managed by Certbot
    ssl_dhparam /etc/letsencrypt/ssl-dhparams.pem; # managed by Certbot

}

server {
    if ($host = www.teqmates.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    if ($host = teqmates.com) {
        return 301 https://$host$request_uri;
    } # managed by Certbot


    listen 80;
    server_name www.teqmates.com teqmates.com;
    return 404; # managed by Certbot
}
